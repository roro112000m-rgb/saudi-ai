    <script>
        const canvas = document.getElementById('drawingLayer');
        const ctx = canvas.getContext('2d');
        const elementsLayer = document.getElementById('elements-layer');
        const contextMenu = document.getElementById('context-menu');
        let currentTool = 'brush';
        let isDrawing = false;
        let activeType = 'image';
        let selectedElement = null;
        let customStickers = [];

        function showToast(msg) {
            const t = document.getElementById('toast');
            t.textContent = msg; t.style.opacity = '1';
            setTimeout(() => t.style.opacity = '0', 3000);
        }

        function openCreateModal() { document.getElementById('create-modal').classList.remove('hidden'); }
        function closeCreateModal() { document.getElementById('create-modal').classList.add('hidden'); }

        function startProject(type) {
            activeType = type;
            closeCreateModal();
            document.getElementById('start-screen').classList.add('hidden');
            document.getElementById('main-app').classList.remove('hidden');
            
            if (type === 'image') {
                document.getElementById('canvas-wrapper').classList.remove('hidden');
                document.getElementById('office-container').classList.add('hidden');
                document.getElementById('image-tools').classList.remove('hidden');
                document.getElementById('doc-tools').classList.add('hidden');
            } else {
                document.getElementById('canvas-wrapper').classList.add('hidden');
                document.getElementById('office-container').classList.remove('hidden');
                document.getElementById('image-tools').classList.add('hidden');
                document.getElementById('doc-tools').classList.remove('hidden');
            }
        }

        function goBack() {
            document.getElementById('main-app').classList.add('hidden');
            document.getElementById('start-screen').classList.remove('hidden');
        }

        function setTool(tool) {
            currentTool = tool;
            document.querySelectorAll('.tool-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(`btn-${tool}`)?.classList.add('active');
            
            if (tool === 'move') {
                canvas.style.pointerEvents = 'none';
                elementsLayer.style.pointerEvents = 'auto';
            } else {
                canvas.style.pointerEvents = 'auto';
                elementsLayer.style.pointerEvents = 'none';
            }
        }

        // --- المصلصقات والكليك يمين ---
        function addSticker(content, isCustom = false, filter = '') {
            if (activeType === 'doc') return;
            const div = document.createElement('div');
            div.className = 'absolute cursor-move select-none text-5xl z-30 pointer-events-auto sticker-item';
            div.style.left = '100px';
            div.style.top = '100px';
            div.style.filter = filter;
            div.innerHTML = content;
            
            div.onmousedown = (e) => { if (e.button === 0) dragElement(e, div); };
            div.oncontextmenu = (e) => {
                e.preventDefault();
                selectedElement = div;
                contextMenu.style.display = 'block';
                contextMenu.style.left = e.pageX + 'px';
                contextMenu.style.top = e.pageY + 'px';
            };
            elementsLayer.appendChild(div);
        }

        window.addEventListener('click', () => { contextMenu.style.display = 'none'; });

        function menuAction(action) {
            if (!selectedElement) return;
            if (action === 'delete') {
                selectedElement.remove();
                showToast('تم الحذف');
            } else if (action === 'copy') {
                const clone = selectedElement.cloneNode(true);
                clone.style.top = (parseInt(selectedElement.style.top) + 20) + 'px';
                clone.style.left = (parseInt(selectedElement.style.left) + 20) + 'px';
                clone.onmousedown = (e) => dragElement(e, clone);
                clone.oncontextmenu = selectedElement.oncontextmenu;
                elementsLayer.appendChild(clone);
                showToast('تم النسخ');
            } else if (action === 'edit') {
                openStickerEditor(selectedElement);
            }
            contextMenu.style.display = 'none';
        }

        // --- تعديل الملصقات ---
        function openStickerEditor(el) {
            const modal = document.getElementById('sticker-editor-modal');
            const preview = document.getElementById('sticker-target-preview');
            preview.innerHTML = el.innerHTML;
            preview.style.filter = el.style.filter;
            modal.classList.remove('hidden');
        }

        function closeStickerEditor() { document.getElementById('sticker-editor-modal').classList.add('hidden'); }

        function applyStickerFilters() {
            const hue = document.getElementById('hue-rotate').value;
            const saturate = document.getElementById('saturate').value;
            const brightness = document.getElementById('brightness').value;
            const preview = document.getElementById('sticker-target-preview');
            preview.style.filter = `hue-rotate(${hue}deg) saturate(${saturate}%) brightness(${brightness}%)`;
        }

        function saveCustomSticker() {
            const preview = document.getElementById('sticker-target-preview');
            const stickerData = { content: preview.innerHTML, filter: preview.style.filter };
            customStickers.push(stickerData);
            updateCustomStickersUI();
            closeStickerEditor();
            showToast('حُفظ في قسم "أنت"');
        }

        function updateCustomStickersUI() {
            const container = document.getElementById('custom-stickers-container');
            container.innerHTML = '';
            customStickers.forEach(s => {
                const btn = document.createElement('button');
                btn.className = 'p-2 bg-gray-800 rounded-lg text-2xl';
                btn.style.filter = s.filter;
                btn.innerHTML = s.content;
                btn.onclick = () => addSticker(s.content, true, s.filter);
                container.appendChild(btn);
            });
        }

        // --- الرسم ---
        canvas.addEventListener('mousedown', (e) => {
            if (currentTool === 'move') return;
            isDrawing = true;
            ctx.beginPath();
            const rect = canvas.getBoundingClientRect();
            ctx.moveTo(e.clientX - rect.left, e.clientY - rect.top);
        });

        canvas.addEventListener('mousemove', (e) => {
            if (!isDrawing) return;
            const rect = canvas.getBoundingClientRect();
            ctx.lineWidth = 8;
            ctx.lineCap = 'round';
            
            if (currentTool === 'eraser') {
                ctx.globalCompositeOperation = 'destination-out';
                ctx.lineWidth = 20;
            } else {
                ctx.globalCompositeOperation = 'source-over';
                ctx.strokeStyle = '#3b82f6';
            }
            
            ctx.lineTo(e.clientX - rect.left, e.clientY - rect.top);
            ctx.stroke();
        });

        window.addEventListener('mouseup', () => isDrawing = false);

        function dragElement(e, el) {
            if (currentTool !== 'move') return;
            let p3 = e.clientX, p4 = e.clientY;
            document.onmousemove = (ev) => {
                let p1 = p3 - ev.clientX; let p2 = p4 - ev.clientY;
                p3 = ev.clientX; p4 = ev.clientY;
                el.style.top = (el.offsetTop - p2) + "px";
                el.style.left = (el.offsetLeft - p1) + "px";
            };
            document.onmouseup = () => { document.onmousemove = null; };
        }

        function insertTable() {
            const table = `<table border="1" style="width:100%; border-collapse:collapse; margin:10px 0;"><tr><td>خلية 1</td><td>خلية 2</td></tr><tr><td>خلية 3</td><td>خلية 4</td></tr></table><p><br></p>`;
            document.execCommand('insertHTML', false, table);
        }

        function saveToDevice() {
            const link = document.createElement('a');
            if (activeType === 'image') {
                link.download = 'dream-design.png';
                link.href = canvas.toDataURL();
            } else {
                const blob = new Blob([document.getElementById('textEditor').innerHTML], {type: 'text/html'});
                link.download = 'dream-doc.html';
                link.href = URL.createObjectURL(blob);
            }
            link.click();
            showToast('تم الحفظ بنجاح!');
        }
    </script>
