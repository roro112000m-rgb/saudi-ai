    <script>
        // تم دمج المفتاح الخاص بك هنا بنجاح
        const apiKey = "AIzaSyBVH5hcfq9-JWizC_L5fXueSQ3eNwt9Ge0"; 
        
        window.onload = () => {
            loadSavedChat();
            setTimeout(() => {
                document.getElementById('introScreen').style.opacity = '0';
                setTimeout(() => document.getElementById('introScreen').style.visibility = 'hidden', 500);
            }, 800);
        };

        function toggleUpdates() { document.getElementById('updatesModal').classList.toggle('active'); document.getElementById('modalOverlay').classList.toggle('active'); }
        function saveChat() { localStorage.setItem('saudi_ai_v2_final_store', document.getElementById('chatHistory').innerHTML); }
        function loadSavedChat() {
            const saved = localStorage.getItem('saudi_ai_v2_final_store');
            if (saved) { document.getElementById('chatHistory').innerHTML = saved; scrollToBottom(); }
            else { addMessage('bot', 'أهلاً بك. أنا جاهز لمساعدتك باللغة العربية واللكجة السعودية.'); }
        }
        function clearChat() { if(confirm("هل تريد مسح كل شيء؟")) { document.getElementById('chatHistory').innerHTML = ""; localStorage.removeItem('saudi_ai_v2_final_store'); addMessage('bot', 'تم البدء من جديد.'); } }
        function toggleDarkMode() { document.documentElement.classList.toggle('dark'); }
        function addMessage(role, text) {
            const container = document.getElementById('chatHistory');
            const div = document.createElement('div');
            div.className = `message-${role}`;
            div.innerHTML = `<div class="message-content">${text}</div>`;
            container.appendChild(div);
            scrollToBottom();
            saveChat();
            return div;
        }
        function scrollToBottom() { const c = document.getElementById('chatHistory'); if (c) c.scrollTop = c.scrollHeight; }
        function startProgressBar(element, duration) {
            let progress = 0;
            const timer = setInterval(() => {
                progress += (50 / duration) * 100;
                if (progress >= 100) { element.style.width = '100%'; clearInterval(timer); }
                else { element.style.width = progress + '%'; }
            }, 50);
        }

        async function askAI() {
            const input = document.getElementById('aiInput');
            const prompt = input.value.trim();
            if (!prompt) return;

            addMessage('user', prompt);
            input.value = "";
            input.style.height = 'auto';

            const isImage = (prompt.includes('صورة') || prompt.includes('ارسم') || prompt.includes('image'));
            const botMsg = addMessage('bot', isImage ? 'جاري إنشاء الصورة...' : 'جاري التفكير...');

            try {
                if (isImage) {
                    botMsg.querySelector('.message-content').innerHTML = `<div class="fast-loader mt-1"><div class="fast-progress" id="p_bar"></div></div>`;
                    startProgressBar(botMsg.querySelector('#p_bar'), 3000);
                    // ملاحظة: Imagen يحتاج إعدادات خاصة، هنا نستخدم رد نصي إذا لم تتوفر صلاحية الصورة
                    botMsg.querySelector('.message-content').innerHTML = "خدمة الصور تتطلب إعدادات إضافية في حساب Google Cloud، لكنني سأجيبك نصياً: جاري معالجة طلبك.";
                } else {
                    const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ contents: [{ parts: [{ text: "أجب بلهجة سعودية مختصرة: " + prompt }] }] })
                    });
                    const data = await res.json();
                    botMsg.querySelector('.message-content').innerHTML = data.candidates?.[0]?.content?.parts?.[0]?.text || "عذراً، لم أستطع الرد.";
                }
            } catch (err) { botMsg.querySelector('.message-content').innerHTML = "حدث خطأ، تأكد من اتصالك."; }
            scrollToBottom();
        }

        const inputField = document.getElementById('aiInput');
        inputField.addEventListener('input', function() { this.style.height = 'auto'; this.style.height = (this.scrollHeight) + 'px'; });
        inputField.addEventListener('keydown', (e) => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); askAI(); } });
    </script>
